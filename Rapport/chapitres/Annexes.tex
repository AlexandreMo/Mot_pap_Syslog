library ieee;
use ieee.std_logic_1164.all;
--use ieee.numeric_std.all;

entity dff_ar is
   port( 
      reset_i   : in    std_logic;
      Clk_i   : in    std_logic;
      D_i     : in    std_logic;
      Q_o     : out   std_logic;
      nQ_o    : out   std_logic
   );
end dff_ar ;

architecture comport of dff_ar is

  signal D_s : std_logic; 
  signal Q_s : std_logic; 

begin
  
   D_s <= D_i; -- Décodeur d'états présent
  
   process(reset_i, Clk_i)
   begin
      if reset_i = '1' then
      Q_s <= '0';
      elsif rising_edge(Clk_i) then
      Q_s <= D_s;
      end if;
   end process;
 
 
   Q_o <= Q_s;
   nQ_o <= not(Q_s);
 end comport;

library ieee;
use ieee.std_logic_1164.all;
--use ieee.numeric_std.all;

entity dff_en is
   port( 
      nReset_i   : in    std_logic;
      Clk_i   : in    std_logic;
      D_i     : in    std_logic;
      EN_i    : in    std_logic;
      Q_o     : out   std_logic;
      nQ_o    : out   std_logic
   );
end dff_en ;

architecture comport of dff_en is
   -- Signaux internes 
  signal reset_s : std_logic;
  signal D_s : std_logic; 
  signal Q_s : std_logic;
begin
   -- Adaptation polarité
   reset_s <= not(nReset_i);
   -- Décodeur d'états présent
   D_s <= D_i when (EN_i = '1') else Q_s;
  
   -- Process de bascule
   process(reset_s, Clk_i)
   begin
      if reset_s = '1' then
      Q_s <= '0'; -- Etat par défaut
      elsif rising_edge(Clk_i) then
      Q_s <= D_s;
      end if;
   end process;
 
   -- Sorties
   Q_o <= Q_s;
   nQ_o <= not(Q_s);
 end comport;

      Q_o     : out    std_logic
   );

  signal D_s : std_logic; -- etat present du latch
  signal reset_s : std_logic; -- etat present du latch
  signal en_s : std_logic; -- etat present du latch
  
  signal Q_s : std_logic;
begin
   -- Decodeur d'etats present
   D_s <= D_i;
   reset_s <= reset_i;
   en_s <= en_i;
   -- Process de bascule
   process(en_s, D_s,reset_s)
   begin
   if reset_s = '1' then
   Q_s <= '0';
   elsif en_s = '1' then
   Q_s <= D_s;
   end if;
   end process;
   --- Sortie
   Q_o <= Q_s;


  
end comport;
